4️. Convert Normal Functions to Arrow Functions
Convert the below functions to arrow functions, then call them with sample values.

function multiply(a, b) {
return a * b;
}
var multiply=(a,b) => a*b
console.log(multiply(2,3))

function greet(name) {
return "Hello "; + name;
}
var greet =(name) => "Hello "+ name;
console.log(greet("Tinu"))

5️. Arrow Function + Array Methods
Given the array:
const numbers = [2, 5, 8, 11, 14];

Create arrow functions to:
1. Return a new array where each number is doubled (map)
var double = numbers.map(num => num *2);
console.log(double)

2. Return only even numbers (filter)
var even =numbers.filter(num => num % 2 ==0);
console.log(even)

3. Return total sum (reduce)
var sum =numbers.reduce((total, num) => total + num, 0);
console.log(sum)
