--Retrieve all employees who work in Sales, Marketing, or IT departments.
select * from Employee where EmpDeptId in (select DeptId from Department where DeptName in ('Sales', 'Marketing','IT'));

--Find all employees with salaries ranging from $50,000 to $75,000 (inclusive).
select * from Employee where EmpSalary between 50000 and 75000;

--List all employees whose last name begins with the letter 'S'.
select * from Employee where EmpLastName like 'S%';

--Display all employees with exactly five letters in their first name.
select * from Employee where EmpFirstName like '_____';

--Find employees whose last name starts with either 'B', 'R', or 'S'.
select * from Employee where EmpLastName like '[B,R,S]%';

--Retrieve all employees whose first name begins with any letter from 'A' through 'M'.
select * from Employee where EmpFirstName like '[A-M]%';

--List employees whose last name doesn't start with a vowel (A, E, I, O, U).
select * from Employee where EmpLastName not like '[A,E,I,O,U]%'; 

--Identify employees earning more than $80,000 annually. 
select * from Employee where EmpSalary*12 > 80000;

--Find employees who joined the company before 2020. 
select * from Employee where Year(HireDate) < 2020;

--List all employees not named 'John' (first name).
select * from Employee where EmpFirstName not in ('John');

--Identify Marketing department employees earning $60,000 or less who were hired after June 30, 2019.
select * from Employee where HireDate > '30-JUN-2019' and EmpSalary < 60000 and EmpDeptId in (select DeptId from Department where DeptName ='Marketing')

--Find employees whose first name contains the letters 'an' anywhere and ends with 'e'.
select * from Employee where EmpFirstName like '%an%e'

--Find employees whose first name contains "%"
select * from Employee where EmpFirstName like '%[%]%'
